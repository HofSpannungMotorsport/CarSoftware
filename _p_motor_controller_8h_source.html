233  pedal_value_t _getPedalPower() {
234  pedal_value_t returnValue = _gasPedal.lastValue;
235 
236  /*
237  Because of FSG Rules, the APPS (gas pedal) has to be locked if
238  - APPS >= 25% || Motor Output >= 5 kW
239  && - Hard Brakeing >= 500ms
240 
241  To Unlock it, the Pedal has to be returned to (regardlessly if brakeing or not)
242  - APPS < 5%
243 
244  All Values are set by defines at the top of this file
245  */
246  if (_brakePedal.lastValue >= _registry.getFloat(PMC_HARD_BRAKE_THRESHHOLD)) {
247  // -> Brake Pedal Position == Hard Brakeing
248  if (_hardBrakeingStarted) {
249  // -> Hard Brakeing already before
250  if (_gasPedalPrimed) {
251  // -> APPS (gas pedal) primed -> active
252  // (otherwise no action needed)
253  if (_hardBrakeingSince.read() >= _registry.getFloat(PMC_HARD_BRAKE_CUTOFF_TIME)) {
254  // -> Hard Brakeing too long -> it is interpreted as a Hard Brake
255  if (_gasPedal.lastValue >= _registry.getFloat(PMC_HARD_BRAKE_CUTOFF_APPS_POSITION)) {
256  // -> Still giving Power throu the Pedal -> Pedal Position too high
257  _gasPedalPrimed = false;
258  }
259 
260  float currentPower = _mapToPowerLimit(returnValue) * _registry.getFloat(PMC_POWER_SET_ON_MC);
261  if (currentPower >= _registry.getFloat(PMC_HARD_BRAKE_CUTOFF_POWER)) {
262  // Calculated Power Output too high -> also unprime the APPS/Gas Pedal
263  _gasPedalPrimed = false;
264  }
265  }
266  }
267  } else {
268  // -> Hard brakeing just started -> start counting...
269  _hardBrakeingStarted = true;
270  _hardBrakeingSince.reset();
271  _hardBrakeingSince.start();
272  }
273  } else {
274  // -> Not hard Brakeing
275  if (_hardBrakeingStarted) {
276  // -> Hard brakeing before, but not anymore -> reset
277  // If APPS (gas pedal) was unprimed before throu this,
278  // it will be unlocked at the next Part of this Method
279  _hardBrakeingStarted = false;
280  }
281  }
282 
283  // Check if pedal is primed, otherwise lock it
284  if (!_gasPedalPrimed) {
285  if (returnValue <= _registry.getFloat(PMC_GAS_PEDAL_PRIME_MIN)) {
286  _gasPedalPrimed = true;
287  } else {
288  returnValue = 0;
289  }
290  }
291 
292  // If brakeing, no current should go to the Motor
293  if (_brakePedal.lastValue >= _registry.getFloat(PMC_BRAKE_POWER_LOCK_THRESHHOLD)) {
294  returnValue = 0;
295  }
296 
297  return returnValue;
298  }

300  float _mapToPowerLimit(float returnValue) {
301  /* We have two values:
302  powerMax: The maximum Power to be applied on our Motor (absolut)
303  powerSetOnController: The Power Maximum set on our Motor Controller.
304 
305  If the Power set on the Motor controller is lower or equal max, it will be limited to the
306  one set on the Motor Controller (obvious). Else, we have to map our value.
307  */
308 
309  float powerMax = _registry.getFloat(PMC_MAX_POWER);
310  float powerSetOnController = _registry.getFloat(PMC_POWER_SET_ON_MC);
311 
312  if (powerMax < powerSetOnController) {
313  returnValue = returnValue * powerMax / powerSetOnController;
314  }
315 
316  return returnValue;
317  }
