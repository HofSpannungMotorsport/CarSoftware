53 FlexCAN::FlexCAN(uint32_t baud)
54 {
55  // set up the pins, 3=PTA12=CAN0_TX, 4=PTA13=CAN0_RX
56  CORE_PIN3_CONFIG = PORT_PCR_MUX(2);
57  CORE_PIN4_CONFIG = PORT_PCR_MUX(2);// | PORT_PCR_PE | PORT_PCR_PS;
58  // select clock source 16MHz xtal
59  OSC0_CR |= OSC_ERCLKEN;
60  SIM_SCGC6 |= SIM_SCGC6_FLEXCAN0;
61  FLEXCAN0_CTRL1 &= ~FLEXCAN_CTRL_CLK_SRC;
62 
63  // enable CAN
64  FLEXCAN0_MCR |= FLEXCAN_MCR_FRZ;
65  FLEXCAN0_MCR &= ~FLEXCAN_MCR_MDIS;
66  while(FLEXCAN0_MCR & FLEXCAN_MCR_LPM_ACK)
67  ;
68  // soft reset
69  FLEXCAN0_MCR ^= FLEXCAN_MCR_SOFT_RST;
70  while(FLEXCAN0_MCR & FLEXCAN_MCR_SOFT_RST)
71  ;
72  // wait for freeze ack
73  while(!(FLEXCAN0_MCR & FLEXCAN_MCR_FRZ_ACK))
74  ;
75  // disable self-reception
76  FLEXCAN0_MCR |= FLEXCAN_MCR_SRX_DIS;
77 
78  //enable RX FIFO
79  FLEXCAN0_MCR |= FLEXCAN_MCR_FEN;
80 
81  // segment splits and clock divisor based on baud rate
82  if ( 250000 == baud ) {
83  FLEXCAN0_CTRL1 = (FLEXCAN_CTRL_PROPSEG(2) | FLEXCAN_CTRL_RJW(1)
84  | FLEXCAN_CTRL_PSEG1(7) | FLEXCAN_CTRL_PSEG2(3) | FLEXCAN_CTRL_PRESDIV(3));
85  } else if ( 500000 == baud ) {
86  FLEXCAN0_CTRL1 = (FLEXCAN_CTRL_PROPSEG(2) | FLEXCAN_CTRL_RJW(1)
87  | FLEXCAN_CTRL_PSEG1(7) | FLEXCAN_CTRL_PSEG2(3) | FLEXCAN_CTRL_PRESDIV(1));
88  } else if ( 1000000 == baud ) {
89  FLEXCAN0_CTRL1 = (FLEXCAN_CTRL_PROPSEG(2) | FLEXCAN_CTRL_RJW(0)
90  | FLEXCAN_CTRL_PSEG1(1) | FLEXCAN_CTRL_PSEG2(1) | FLEXCAN_CTRL_PRESDIV(1));
91  } else { // 125000
92  FLEXCAN0_CTRL1 = (FLEXCAN_CTRL_PROPSEG(2) | FLEXCAN_CTRL_RJW(1)
93  | FLEXCAN_CTRL_PSEG1(7) | FLEXCAN_CTRL_PSEG2(3) | FLEXCAN_CTRL_PRESDIV(7));
94  }
95 
96  // Default mask is allow everything
97  defaultMask.rtr = 0;
98  defaultMask.ext = 0;
99  defaultMask.id = 0;
100 }

114 void FlexCAN::begin(const CAN_filter_t &mask)
115 {
116  FLEXCAN0_RXMGMASK = 0;
117 
118  //enable reception of all messages that fit the mask
119  if (mask.ext) {
120  FLEXCAN0_RXFGMASK = ((mask.rtr?1:0) << 31) | ((mask.ext?1:0) << 30) | ((mask.id & FLEXCAN_MB_ID_EXT_MASK) << 1);
121  } else {
122  FLEXCAN0_RXFGMASK = ((mask.rtr?1:0) << 31) | ((mask.ext?1:0) << 30) | (FLEXCAN_MB_ID_IDSTD(mask.id) << 1);
123  }
124 
125  // start the CAN
126  FLEXCAN0_MCR &= ~(FLEXCAN_MCR_HALT);
127  // wait till exit of freeze mode
128  while(FLEXCAN0_MCR & FLEXCAN_MCR_FRZ_ACK);
129 
130  // wait till ready
131  while(FLEXCAN0_MCR & FLEXCAN_MCR_NOT_RDY);
132 
133  //set tx buffers to inactive
134  for (int i = txb; i < txb + txBuffers; i++) {
135  FLEXCAN0_MBn_CS(i) = FLEXCAN_MB_CS_CODE(FLEXCAN_MB_CODE_TX_INACTIVE);
136  }
137 }

162 int FlexCAN::read(CAN_message_t &msg)
163 {
164  unsigned long int startMillis;
165 
166  startMillis = msg.timeout? millis() : 0;
167 
168  while( !available() ) {
169  if ( !msg.timeout || (msg.timeout<=(millis()-startMillis)) ) {
170  // early EXIT nothing here
171  return 0;
172  }
173  yield();
174  }
175 
176  // get identifier and dlc
177  msg.len = FLEXCAN_get_length(FLEXCAN0_MBn_CS(rxb));
178  msg.ext = (FLEXCAN0_MBn_CS(rxb) & FLEXCAN_MB_CS_IDE)? 1:0;
179  msg.id = (FLEXCAN0_MBn_ID(rxb) & FLEXCAN_MB_ID_EXT_MASK);
180  if(!msg.ext) {
181  msg.id >>= FLEXCAN_MB_ID_STD_BIT_NO;
182  }
183 
184  // copy out message
185  uint32_t dataIn = FLEXCAN0_MBn_WORD0(rxb);
186  msg.buf[3] = dataIn;
187  dataIn >>=8;
188  msg.buf[2] = dataIn;
189  dataIn >>=8;
190  msg.buf[1] = dataIn;
191  dataIn >>=8;
192  msg.buf[0] = dataIn;
193  if ( 4 < msg.len ) {
194  dataIn = FLEXCAN0_MBn_WORD1(rxb);
195  msg.buf[7] = dataIn;
196  dataIn >>=8;
197  msg.buf[6] = dataIn;
198  dataIn >>=8;
199  msg.buf[5] = dataIn;
200  dataIn >>=8;
201  msg.buf[4] = dataIn;
202  }
203  for( int loop=msg.len; loop<8; ++loop ) {
204  msg.buf[loop] = 0;
205  }
206 
207  //notify FIFO that message has been read
208  FLEXCAN0_IFLAG1 = FLEXCAN_IMASK1_BUF5M;
209 
210  return 1;
211 }

215 int FlexCAN::write(const CAN_message_t &msg)
216 {
217  unsigned long int startMillis;
218 
219  startMillis = msg.timeout? millis() : 0;
220 
221  // find an available buffer
222  int buffer = -1;
223  for ( int index = txb; ; ) {
224  if ((FLEXCAN0_MBn_CS(index) & FLEXCAN_MB_CS_CODE_MASK) == FLEXCAN_MB_CS_CODE(FLEXCAN_MB_CODE_TX_INACTIVE)) {
225  buffer = index;
226  break;// found one
227  }
228  if ( !msg.timeout ) {
229  if ( ++index >= (txb+txBuffers) ) {
230  return 0;// early EXIT no buffers available
231  }
232  } else {
233  // blocking mode, only 1 txb used to guarantee frames in order
234  if ( msg.timeout <= (millis()-startMillis) ) {
235  return 0;// timed out
236  }
237  yield();
238  }
239  }
240 
241  // transmit the frame
242  FLEXCAN0_MBn_CS(buffer) = FLEXCAN_MB_CS_CODE(FLEXCAN_MB_CODE_TX_INACTIVE);
243  if(msg.ext) {
244  FLEXCAN0_MBn_ID(buffer) = (msg.id & FLEXCAN_MB_ID_EXT_MASK);
245  } else {
246  FLEXCAN0_MBn_ID(buffer) = FLEXCAN_MB_ID_IDSTD(msg.id);
247  }
248  FLEXCAN0_MBn_WORD0(buffer) = (msg.buf[0]<<24)|(msg.buf[1]<<16)|(msg.buf[2]<<8)|msg.buf[3];
249  FLEXCAN0_MBn_WORD1(buffer) = (msg.buf[4]<<24)|(msg.buf[5]<<16)|(msg.buf[6]<<8)|msg.buf[7];
250  if(msg.ext) {
251  FLEXCAN0_MBn_CS(buffer) = FLEXCAN_MB_CS_CODE(FLEXCAN_MB_CODE_TX_ONCE)
252  | FLEXCAN_MB_CS_LENGTH(msg.len) | FLEXCAN_MB_CS_SRR | FLEXCAN_MB_CS_IDE;
253  } else {
254  FLEXCAN0_MBn_CS(buffer) = FLEXCAN_MB_CS_CODE(FLEXCAN_MB_CODE_TX_ONCE)
255  | FLEXCAN_MB_CS_LENGTH(msg.len);
256  }
257 
258  return 1;
259 }
